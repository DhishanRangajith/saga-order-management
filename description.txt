CREATE TABLE payments (
    id UUID PRIMARY KEY,
    order_id UUID NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    status VARCHAR(20) NOT NULL,  -- e.g., PENDING, SUCCESS, FAILED, CANCELL
    created_at, updated_at
);
